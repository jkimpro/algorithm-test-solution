#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
#include <vector>

using namespace std;

int n;
long long result;
long long arr[100][100];
bool member[100];

vector <int> team1;
vector <int> team2;

void dfs(int count, int startNum) {
	//ÆÀ ºÐ·ù
	if (count == (n / 2)) {
		long long score;
		long long score1 = 0;
		long long score2 = 0;

		for (int i = 0; i < n; i++) {
			if (member[i] == true) {
				team1.push_back(i);
			}
			else {
				team2.push_back(i);
			}
		}
		//team1 °è»ê
		for (int i = 0; i < team1.size(); i++) {
			for (int j = 0; j < team1.size(); j++) {
				if (i == j) { continue; }
				else {
					score1 += arr[team1[i]][team1[j]];
					score2 += arr[team2[i]][team2[j]];
				}
			}
		}
		score = abs(score1 - score2);
		result = min(result, score);

		team1.clear();
		team2.clear();
		return;
	}
	else {
		for (int i = startNum; i < n; i++) {
			if (member[i] == false) {
				member[i] = true;
				dfs(count + 1, i+1);
				member[i] = false;
			}
		}

	}
	
}
int main() {
	result = 99999999999;
	scanf("%d", &n);

	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			scanf("%lld", &arr[i][j]);
		}
	}
	for (int i = 0; i < n; i++) {
		member[i] = false;
	}

	dfs(0,0);
	
	printf("%lld", result);

	return 0;
}
