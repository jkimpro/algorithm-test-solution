	bool go = false;
	scanf("%s", arr);

	for (int i = 0; ; i++) {
		if (arr[i] == '\0') { break; }
		else {
			Atom temp;
			if (arr[i] >= 'A' && arr[i] <= 'Z') {
				temp.name[0] = arr[i];
				
				if (arr[i + 1] >= 'a' && arr[i + 1] <= 'z') {
					temp.name[1] = arr[i + 1];
					i += 1;
				}
				else {
					temp.name[1] = '\0';
				}

				temp.num = -1;
				atoms.push_back(temp);
			}
			else { bar = i; break; }
		}
	}
	

	for(int i = bar; ; i++) {
		if (arr[i] == '\0') { break; }
		else if (vecBar >= atoms.size()){
			printf("error");
			return 0;
		}
		else {
			//printf("\narr[i] = %c, vecBar=%d\n", arr[i],vecBar);
			if (arr[i] >= '1' && arr[i] <= '9') {
				atoms[vecBar].num = arr[i] - '0';
				if (arr[i] == '1' && arr[i + 1] == '0') {
					atoms[vecBar].num *= 10;
					i += 1;
				}
				vecBar++;
			}
		}
	}
	//확인
	for (int i = 0; i < atoms.size(); i++) {
		if (atoms[i].num == -1) {
			printf("error");
			return 0;
		}
	}
	
	for (int i = 0; i < atoms.size(); i++) {
		printf("%c", atoms[i].name[0]);
		if (atoms[i].name[1] != '\0') {
			printf("%c", atoms[i].name[1]);
		}
		if (atoms[i].num == 1) {
			continue;
		}
		else {
			printf("%d", atoms[i].num);
		}
	
	}
	return 0;
}
